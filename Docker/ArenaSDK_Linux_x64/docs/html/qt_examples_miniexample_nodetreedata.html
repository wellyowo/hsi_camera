#pragma once
#include <QtWidgets>
#include "ArenaApi.h"
#include "SaveApi.h"

#include "NodeTreeItem.h"

class NodeTreeData : public QStyledItemDelegate
{

public:
        NodeTreeData(QWidget* pParent) :
                QStyledItemDelegate(pParent)
        {
        }

        ~NodeTreeData()
        {
        }

        QWidget* createEditor(QWidget* pParent, const QStyleOptionViewItem& option, const QModelIndex& index) const override
        {
                if (!index.isValid())
                {
                        return NULL;
                }

                NodeTreeItem* pItem = static_cast<NodeTreeItem*>(index.internalPointer());
                pItem->settingValue();
                GenApi::EInterfaceType nodeType = pItem->getNode()->GetPrincipalInterfaceType();

                QWidget* pEditorWidget = NULL;

                switch (nodeType)
                {
                        case GenApi::intfIBoolean: {
                                QComboBox* pCheckBox = new QComboBox(pParent);
                                pEditorWidget = pCheckBox;
                                break;
                        }
                        case GenApi::intfICommand: {
                                QPushButton* pPushButton = new QPushButton(pParent);
                                pEditorWidget = pPushButton;
                                break;
                        }
                        case GenApi::intfIEnumeration: {
                                QComboBox* pComboBox = new QComboBox(pParent);
                                pEditorWidget = pComboBox;
                                break;
                        }
                        case GenApi::intfIFloat: {
                                QDoubleSpinBox* pDoubleSpinBox = new QDoubleSpinBox(pParent);
                                pEditorWidget = pDoubleSpinBox;
                                break;
                        }
                        case GenApi::intfIInteger: {
                                QSpinBox* pSpinBox = new QSpinBox(pParent);
                                pEditorWidget = pSpinBox;
                                break;
                        }
                        case GenApi::intfIString: {
                                QLineEdit* pLineEdit = new QLineEdit(pParent);
                                pEditorWidget = pLineEdit;
                                break;
                        }
                }

                return pEditorWidget;
        }

        void setEditorData(QWidget* pEditor, const QModelIndex& index) const override
        {
                if (!index.isValid())
                {
                        return;
                }

                NodeTreeItem* pItem = static_cast<NodeTreeItem*>(index.internalPointer());
                GenApi::EInterfaceType nodeType = pItem->getNode()->GetPrincipalInterfaceType();
                try
                {
                        switch (nodeType)
                        {
                                case GenApi::intfIBoolean: {
                                        GenApi::CBooleanPtr pBoolean = pItem->getNode();
                                        QComboBox* pCheckBox = static_cast<QComboBox*>(pEditor);
                                        bool x = pBoolean->GetValue();
                                        pCheckBox->addItem(x ? "true" : "false");
                                        pCheckBox->addItem(!x ? "true" : "false");
                                        break;
                                }
                                case GenApi::intfICommand: {
                                        QPushButton* pPushButton = static_cast<QPushButton*>(pEditor);
                                        pPushButton->setText("Execute");
                                        break;
                                }
                                case GenApi::intfIEnumeration: {
                                        GenApi::CEnumerationPtr pEnumeration = pItem->getNode();
                                        QComboBox* pComboBox = static_cast<QComboBox*>(pEditor);
                                        GenApi::NodeList_t entries;
                                        pEnumeration->GetEntries(entries);

                                        for (GenApi::CEnumEntryPtr pEnumEntry : entries)
                                        {
                                                pComboBox->addItem(QString(pEnumEntry->GetSymbolic()));
                                        }

                                        GenApi::CEnumEntryPtr pCurrentEnumEntry = pEnumeration->GetCurrentEntry();
                                        pComboBox->setCurrentText(QString(pCurrentEnumEntry->GetSymbolic()));
                                        break;
                                }
                                case GenApi::intfIFloat: {
                                        GenApi::CFloatPtr pFloat = pItem->getNode();
                                        QDoubleSpinBox* pDoubleSpinBox = static_cast<QDoubleSpinBox*>(pEditor);
                                        pDoubleSpinBox->setMinimum(pFloat->GetMin());
                                        pDoubleSpinBox->setSingleStep(1);
                                        pDoubleSpinBox->setMaximum(pFloat->GetMax());
                                        int value = (pFloat->GetValue());
                                        pDoubleSpinBox->setValue(value);
                                        break;
                                }
                                case GenApi::intfIInteger: {
                                        GenApi::CIntegerPtr pInteger = pItem->getNode();
                                        QSpinBox* pSpinBox = static_cast<QSpinBox*>(pEditor);
                                        pSpinBox->setMinimum(pInteger->GetMin());
                                        pSpinBox->setSingleStep(pInteger->GetInc());
                                        pSpinBox->setMaximum(pInteger->GetMax());
                                        pSpinBox->setValue(pInteger->GetValue());
                                        break;
                                }
                                case GenApi::intfIString: {
                                        GenApi::CStringPtr pString = pItem->getNode();
                                        QLineEdit* pLineEdit = static_cast<QLineEdit*>(pEditor);
                                        pLineEdit->setText(QString(pString->GetValue()));
                                        break;
                                }
                        }
                }
                catch (...)
                {
                }
        }

        void setModelData(QWidget* pEditor, QAbstractItemModel* pModel, const QModelIndex& index) const override
        {
                if (!index.isValid())
                {
                        return;
                }
                QVariant value = "";
                try
                {
                        NodeTreeItem* pItem = static_cast<NodeTreeItem*>(index.internalPointer());
                        GenApi::EInterfaceType nodeType = pItem->getNode()->GetPrincipalInterfaceType();

                        switch (nodeType)
                        {
                                case GenApi::intfIBoolean: {
                                        QComboBox* pCheckBox = static_cast<QComboBox*>(pEditor);
                                        value = pCheckBox->currentText();
                                        break;
                                }
                                case GenApi::intfICommand: {
                                        // execute without value
                                        break;
                                }
                                case GenApi::intfIEnumeration: {
                                        QComboBox* pComboBox = static_cast<QComboBox*>(pEditor);
                                        value = pComboBox->currentText();
                                        break;
                                }
                                case GenApi::intfIFloat: {
                                        GenApi::CFloatPtr pFloat = pItem->getNode();
                                        QDoubleSpinBox* pDoubleSpinBox = static_cast<QDoubleSpinBox*>(pEditor);
                                        value = static_cast<double>(pDoubleSpinBox->value());
                                        break;
                                }
                                case GenApi::intfIInteger: {
                                        QSpinBox* pSpinBox = static_cast<QSpinBox*>(pEditor);
                                        value = pSpinBox->value();
                                        break;
                                }
                                case GenApi::intfIString: {
                                        QLineEdit* pLineEdit = static_cast<QLineEdit*>(pEditor);
                                        value = pLineEdit->text();
                                        break;
                                }
                        }
                }
                catch (...)
                {
                }

                pModel->setData(index, value, Qt::EditRole);
        }

        void updateEditorGeometry(QWidget* pEditor, const QStyleOptionViewItem& option, const QModelIndex& index) const override
        {
                pEditor->setGeometry(option.rect);
        }
};

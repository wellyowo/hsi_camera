#pragma once
#include <QApplication>
#include "stdafx.h"
#include "ArenaApi.h"
#include "SaveApi.h"

class ArenaManager : public QObject
{

private:
        Arena::ISystem* m_pSystem = NULL;
        Arena::IDevice* m_pDevice = NULL;
        std::vector<std::string> m_devices;

public:
        ArenaManager()
        {
                // open the system on start
                m_pSystem = Arena::OpenSystem();
        }

        ~ArenaManager()
        {
                // destroy device and close system
                if (m_pDevice)
                {
                        m_pSystem->DestroyDevice(m_pDevice);
                }
                Arena::CloseSystem(m_pSystem);
        }

        // get device list
        virtual std::vector<std::string> GetDeviceList()
        {

                if (m_pSystem->UpdateDevices(100))
                {
                        m_devices.clear();

                        // create list containing device information
                        std::vector<Arena::DeviceInfo> devices = m_pSystem->GetDevices();
                        for (size_t i = 0; i < devices.size(); i++)
                        {
                                Arena::DeviceInfo device = devices.at(i);
                                std::string deviceId = device.VendorName() + " " + device.ModelName() + ": Serial " + device.SerialNumber();
                                m_devices.push_back(deviceId);
                        }
                }
                return m_devices;
        }

        // select device
        virtual bool SelectDevice(std::string selectedDevice)
        {
                if (m_pDevice)
                {
                        GenApi::CValuePtr pSerialNumber = m_pDevice->GetTLDeviceNodeMap()->GetNode("DeviceSerialNumber");
                        if (selectedDevice.find(pSerialNumber->ToString()) != std::string::npos)
                        {
                                // device already selected
                                return true;
                        }

                        // destroy any previously selected device
                        m_pSystem->DestroyDevice(m_pDevice);
                        m_pDevice = NULL;
                }

                std::vector<Arena::DeviceInfo> deviceInfos = m_pSystem->GetDevices();

                // no devices connected
                if (deviceInfos.size() == 0)
                        return false;

                for (size_t i = 0; i < deviceInfos.size(); i++)
                {
                        if (selectedDevice.find(deviceInfos.at(i).SerialNumber()) != std::string::npos)
                        {
                                try
                                {
                                        // create device
                                        m_pDevice = m_pSystem->CreateDevice(deviceInfos.at(i));
                                        return true;
                                }
                                catch (...)
                                {
                                        // unable to create device
                                        return false;
                                }
                        }
                }
        }

        // disconnect device
        virtual void DisconnectDevice()
        {
                if (m_pDevice)
                {
                        m_pSystem->DestroyDevice(m_pDevice);
                        m_pDevice = NULL;
                }
        }
};

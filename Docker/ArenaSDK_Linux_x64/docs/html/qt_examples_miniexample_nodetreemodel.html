#pragma once
#include <QtCore>
#include "ArenaManager.h"
#include "NodeTreeItem.h"

class NodeTreeModel : public QAbstractItemModel
{
        Q_OBJECT

private:
        ArenaManager& m_arenaManager;
        NodeTreeItem* m_pRoot;
        std::vector<intptr_t> m_hEvents;
        bool m_isCallback = false;
        GenApi::INodeMap* m_pNodeMap = NULL;

public:
        NodeTreeModel(ArenaManager& arenaManager, QObject* pParent) :
                m_arenaManager(arenaManager),
                QAbstractItemModel(pParent)
        {
                // set up node tree
                m_pNodeMap = arenaManager.GetNodeMap();
                setUpTree(m_pNodeMap);

                // Timer to update polled nodes every second
                QTimer* pollTimer = new QTimer(this);
                connect(pollTimer, SIGNAL(timeout()), this, SLOT(PollNodeMap()));
                pollTimer->start(1000);
        }

        ~NodeTreeModel()
        {
                while (m_hEvents.size() > 0)
                {
                        GenApi::Deregister(m_hEvents.back());
                        m_hEvents.pop_back();
                }

                delete m_pRoot;
        }

        virtual QModelIndex index(int row, int column, const QModelIndex& parent = QModelIndex()) const
        {
                if (!hasIndex(row, column, parent))
                {
                        return QModelIndex();
                }

                NodeTreeItem* pParent = NULL;

                if (!parent.isValid())
                {
                        pParent = m_pRoot;
                }
                else
                {
                        pParent = static_cast<NodeTreeItem*>(parent.internalPointer());
                }

                NodeTreeItem* childItem = pParent->child(row);

                if (childItem)
                {
                        return createIndex(row, column, childItem);
                }
                else
                {
                        return QModelIndex();
                }
        }

        virtual QModelIndex parent(const QModelIndex& index) const
        {
                if (!index.isValid())
                {
                        return QModelIndex();
                }

                NodeTreeItem* pChild = static_cast<NodeTreeItem*>(index.internalPointer());
                NodeTreeItem* pParent = pChild->parentItem();

                if (pParent == m_pRoot)
                {
                        return QModelIndex();
                }

                return createIndex(pParent->row(), 0, pParent);
        }

        virtual int rowCount(const QModelIndex& parent = QModelIndex()) const
        {
                NodeTreeItem* pParent;

                if (parent.column() > 0)
                {
                        return 0;
                }

                if (!parent.isValid())
                {
                        pParent = m_pRoot;
                }
                else
                {
                        pParent = static_cast<NodeTreeItem*>(parent.internalPointer());
                }

                return pParent->childCount();
        }

        virtual int columnCount(const QModelIndex& parent = QModelIndex()) const
        {
                if (parent.isValid())
                {
                        return static_cast<NodeTreeItem*>(parent.internalPointer())->columnCount();
                }
                else
                {
                        return m_pRoot->columnCount();
                }
        }

        virtual QVariant data(const QModelIndex& index, int role = Qt::DisplayRole) const
        {
                if (!index.isValid())
                {
                        return QVariant();
                }

                if (role != Qt::DisplayRole)
                {
                        return QVariant();
                }

                NodeTreeItem* pItem = static_cast<NodeTreeItem*>(index.internalPointer());

                return pItem->data(index.column());
        }

        virtual QVariant headerData(int section, Qt::Orientation orientation, int role = Qt::DisplayRole) const override
        {
                if (orientation == Qt::Horizontal && role == Qt::DisplayRole)
                {
                        return m_pRoot->data(section);
                }

                return QVariant();
        }

        virtual Qt::ItemFlags flags(const QModelIndex& index) const override
        {
                if (!index.isValid())
                {
                        return 0;
                }

                if (index.column() == 1)
                {
                        return QAbstractItemModel::flags(index) | Qt::ItemFlag::ItemIsEditable;
                }

                return QAbstractItemModel::flags(index);
        }

        virtual bool setData(const QModelIndex& index, const QVariant& value, int role = Qt::EditRole)
        {
                if (!index.isValid())
                {
                        return false;
                }

                NodeTreeItem* pItem = static_cast<NodeTreeItem*>(index.internalPointer());
                return pItem->setValue(value, m_isCallback);
        }

public slots:
        void PollNodeMap()
        {
                m_pNodeMap->Poll(1000);
        }

private:
        void setUpTree(GenApi::INodeMap* pNodeMap)
        {
                // get root node
                GenApi::CCategoryPtr pRoot = pNodeMap->GetNode("Root");

                // create root item
                QList<QVariant> rootData;
                rootData << "Node"
                                 << "Value";
                m_pRoot = new NodeTreeItem(rootData, NULL, pRoot->GetNode());

                // get root children
                GenApi::FeatureList_t children;
                pRoot->GetFeatures(children);

                // add children to root
                for (GenApi::IValue* pChild : children)
                {
                        addNode(pChild->GetNode(), m_pRoot);
                }
        }

        // retrieve the node value on callback to be displayed
        void NodeTreeModel::OnNodeCallback(GenApi::INode* pNode)
        {
                m_isCallback = true;
                QString nodeName = pNode->GetDisplayName();

                // returns a list of indexes in that column that match the specified value
                QModelIndexList foundIndex = this->match(index(0, 0),
                                                                                                 Qt::DisplayRole,
                                                                                                 nodeName,
                                                                                                 1,
                                                                                                 Qt::MatchRecursive);

                // retrieve the node's new value from the device
                GenApi::EInterfaceType nodeType = pNode->GetPrincipalInterfaceType();
                QVariant nodeValue = "";

                if (GenApi::IsReadable(pNode))
                {
                        try
                        {
                                switch (nodeType)
                                {
                                        case GenApi::intfIBoolean: {
                                                GenApi::CBooleanPtr pBoolean = pNode;
                                                nodeValue = pBoolean->GetValue();
                                                break;
                                        }
                                        case GenApi::intfICommand: {
                                                nodeValue = "";
                                                break;
                                        }
                                        case GenApi::intfIEnumeration: {
                                                GenApi::CEnumerationPtr pEnumeration = pNode;
                                                GenApi::CEnumEntryPtr pEnumEntry = pEnumeration->GetCurrentEntry();
                                                nodeValue = QString(pEnumEntry->GetSymbolic());
                                                break;
                                        }
                                        case GenApi::intfIFloat: {
                                                GenApi::CFloatPtr pFloat = pNode;
                                                nodeValue = pFloat->GetValue();
                                                break;
                                        }
                                        case GenApi::intfIInteger: {
                                                GenApi::CIntegerPtr pInteger = pNode;
                                                nodeValue = pInteger->GetValue();
                                                break;
                                        }
                                        case GenApi::intfIString: {
                                                GenApi::CStringPtr pString = pNode;
                                                nodeValue = QString(pString->GetValue());
                                                break;
                                        }
                                }
                        }
                        catch (...)
                        {
                        }

                        // set the new data to the viewer
                        setData(index(foundIndex[0].row(), foundIndex[0].column() + 1, foundIndex[0].parent()), nodeValue, Qt::EditRole);

                        // exiting callback
                        m_isCallback = false;
                }
        }

        void addNode(GenApi::INode* pNode, NodeTreeItem* pParent = NULL)
        {

                // create data for node tree item
                QList<QVariant> data;

                // add node name as column 1
                QString nodeName = pNode->GetDisplayName();
                data.push_back(nodeName);

                // add value as column 2 depending on type
                GenApi::EInterfaceType nodeType = pNode->GetPrincipalInterfaceType();
                QVariant nodeValue = "";

                if (GenApi::IsReadable(pNode))
                {
                        switch (nodeType)
                        {
                                case GenApi::intfIBoolean: {
                                        GenApi::CBooleanPtr pBoolean = pNode;
                                        nodeValue = pBoolean->GetValue();
                                        break;
                                }
                                case GenApi::intfICommand: {
                                        nodeValue = "";
                                        break;
                                }
                                case GenApi::intfIEnumeration: {
                                        GenApi::CEnumerationPtr pEnumeration = pNode;
                                        GenApi::CEnumEntryPtr pEnumEntry = pEnumeration->GetCurrentEntry();
                                        nodeValue = QString(pEnumEntry->GetSymbolic());
                                        break;
                                }
                                case GenApi::intfIFloat: {
                                        GenApi::CFloatPtr pFloat = pNode;
                                        nodeValue = pFloat->GetValue();
                                        break;
                                }
                                case GenApi::intfIInteger: {
                                        GenApi::CIntegerPtr pInteger = pNode;
                                        nodeValue = pInteger->GetValue();
                                        break;
                                }
                                case GenApi::intfIString: {
                                        GenApi::CStringPtr pString = pNode;
                                        nodeValue = QString(pString->GetValue());
                                        break;
                                }
                        }
                }

                data.push_back(nodeValue);

                // create node tree item
                NodeTreeItem* pNodeTreeItem = new NodeTreeItem(data, pParent, pNode);

                // add children to node tree item if applicable
                if (pNode->GetPrincipalInterfaceType() == GenApi::intfICategory)
                {
                        GenApi::CCategoryPtr pCategory = pNode;

                        GenApi::FeatureList_t children;
                        pCategory->GetFeatures(children);

                        for (GenApi::IValue* pChild : children)
                        {
                                addNode(pChild->GetNode(), pNodeTreeItem);
                        }
                }

                intptr_t hEvent = GenApi::Register(pNode, *this, &NodeTreeModel::OnNodeCallback);
                m_hEvents.push_back(hEvent);

                pParent->AppendChild(pNodeTreeItem);
        }
};

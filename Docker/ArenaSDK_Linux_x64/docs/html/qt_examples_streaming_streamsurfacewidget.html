#pragma once
#include "StreamSurface.h"
#include "QtWidgets/qwidget.h"
#include "QtGui/qevent.h"
#include "QtCore/qrect.h"
#include "QtGui/qimage.h"
#include "QtMultimedia/qabstractvideosurface.h"
#include "QtMultimedia/qvideoframe.h"
#include "QtMultimedia/qvideosurfaceformat.h"
#include "QtGui/qpainter.h"

class StreamSurfaceWidget : public QWidget
{
        Q_OBJECT

private:
        StreamSurface* m_pSurface;

public:
        StreamSurfaceWidget(QWidget* parent = 0) :
                QWidget(parent),
                m_pSurface(NULL)
        {
                setAutoFillBackground(false);
                setAttribute(Qt::WA_NoSystemBackground, true);
                setAttribute(Qt::WA_PaintOnScreen, true);

                QPalette palette = this->palette();
                palette.setColor(QPalette::Background, Qt::black);
                setPalette(palette);

                setSizePolicy(QSizePolicy::MinimumExpanding, QSizePolicy::MinimumExpanding);

                m_pSurface = new StreamSurface(this);
        }

        ~StreamSurfaceWidget()
        {
                delete m_pSurface;
        }

        QAbstractVideoSurface* videoSurface() const
        {
                return m_pSurface;
        }

        QSize sizeHint() const
        {
                return m_pSurface->surfaceFormat().sizeHint();
        }

protected:
        void paintEvent(QPaintEvent* pEvent)
        {
                QPainter painter(this);

                if (m_pSurface->isActive())
                {
                        const QRect videoRect = m_pSurface->videoRect();

                        if (!videoRect.contains(pEvent->rect()))
                        {
                                QRegion region = pEvent->region();
                                QBrush brush = palette().background();

                                foreach (const QRect& rect, region.rects())
                                        painter.fillRect(rect, brush);
                        }

                        m_pSurface->paint(&painter);
                }
                else
                {
                        painter.fillRect(pEvent->rect(), palette().background());
                }
        }

        void resizeEvent(QResizeEvent* pEvent)
        {
                QWidget::resizeEvent(pEvent);
                m_pSurface->updateVideoRect();
        }
};

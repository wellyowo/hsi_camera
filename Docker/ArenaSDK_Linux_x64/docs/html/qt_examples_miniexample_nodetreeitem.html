#pragma once
#include "ArenaApi.h"
#include "SaveApi.h"
#include <QtCore>

class NodeTreeItem
{
private:
        QList<NodeTreeItem*> m_children;
        QList<QVariant> m_data;
        NodeTreeItem* m_pParent;
        GenApi::INode* m_pNode;
        QVariant m_tmpWhileSetting;

public:
        NodeTreeItem(const QList<QVariant>& data, NodeTreeItem* pParent, GenApi::INode* pNode) :
                m_data(data),
                m_pParent(pParent),
                m_pNode(pNode)
        {
        }

        ~NodeTreeItem()
        {
                qDeleteAll(m_children);
        }

        void AppendChild(NodeTreeItem* pChild)
        {
                m_children.append(pChild);
        }

        NodeTreeItem* child(int row)
        {
                return m_children.at(row);
        }

        int childCount() const
        {
                return m_children.size();
        }

        int columnCount() const
        {
                return m_data.size();
        }

        QVariant data(int column) const
        {
                return m_data.at(column);
        }

        int row() const
        {
                if (m_pParent)
                {
                        return m_pParent->m_children.indexOf(const_cast<NodeTreeItem*>(this));
                }

                return 0;
        }

        NodeTreeItem* parentItem() const
        {
                return m_pParent;
        }

        GenApi::INode* getNode() const
        {
                return m_pNode;
        }

        void settingValue()
        {
                m_tmpWhileSetting = m_data[1];
                m_data[1] = "";
        }

        void returnValue()
        {
                m_data[1] = m_tmpWhileSetting;
        }

        bool setValue(QVariant valueData, bool isCallback)
        {
                try
                {
                        if (!isCallback)
                        {
                                GenApi::EInterfaceType nodeType = m_pNode->GetPrincipalInterfaceType();

                                switch (nodeType)
                                {
                                        case GenApi::intfIBoolean: {
                                                GenApi::CBooleanPtr pBoolean = m_pNode;
                                                bool value = valueData.toBool();
                                                pBoolean->SetValue(value);
                                                break;
                                        }
                                        case GenApi::intfICommand: {
                                                GenApi::CCommandPtr pCommand = m_pNode;
                                                pCommand->Execute();
                                                break;
                                        }
                                        case GenApi::intfIEnumeration: {
                                                GenApi::CEnumerationPtr pEnumeration = m_pNode;
                                                std::string symbolic = valueData.toString().toStdString();
                                                GenApi::CEnumEntryPtr pEnumEntry = pEnumeration->GetEntryByName(symbolic.c_str());
                                                int64_t value = pEnumEntry->GetValue();
                                                pEnumeration->SetIntValue(value);
                                                break;
                                        }
                                        case GenApi::intfIFloat: {
                                                GenApi::CFloatPtr pFloat = m_pNode;
                                                double value = valueData.toDouble();

                                                if (value < pFloat->GetMin())
                                                {
                                                        value = pFloat->GetMin();
                                                }

                                                if (value > pFloat->GetMax())
                                                {
                                                        value = pFloat->GetMax();
                                                }

                                                pFloat->SetValue(value);
                                                break;
                                        }
                                        case GenApi::intfIInteger: {
                                                GenApi::CIntegerPtr pInteger = m_pNode;
                                                int64_t value = valueData.toLongLong();
                                                pInteger->SetValue(value);
                                                break;
                                        }
                                        case GenApi::intfIString: {
                                                GenApi::CStringPtr pString = m_pNode;
                                                std::string value = valueData.toString().toStdString();
                                                pString->SetValue(value.c_str());
                                                break;
                                        }
                                }
                        }

                        // update value in view/model
                        m_data[1] = valueData;

                        return true;
                }
                catch (...)
                {
                }

                returnValue();
                return false;
        }
};

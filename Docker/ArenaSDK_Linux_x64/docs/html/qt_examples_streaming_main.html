#include <QtWidgets/QApplication>
#include "DeviceWidget.h"
#include "ArenaManager.h"

int main(int argc, char* argv[])
{
        QApplication a(argc, argv);
        ArenaManager arenaManager;
        DeviceWidget deviceWidget(arenaManager);
        deviceWidget.show();
        a.exec();
        arenaManager.quit();
        arenaManager.wait();
        return 0;
}

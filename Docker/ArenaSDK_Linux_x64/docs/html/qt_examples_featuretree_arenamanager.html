#pragma once
#include <QApplication>
#include "stdafx.h"
#include "ArenaApi.h"
#include "SaveApi.h"

class ArenaManager : public QObject
{

private:
        Arena::ISystem* m_pSystem = NULL;
        Arena::IDevice* m_pDevice = NULL;
        std::string m_device = "";
        std::vector<Arena::DeviceInfo> m_deviceList;

public:
        ArenaManager()
        {
                // open the system on start
                m_pSystem = Arena::OpenSystem();
        }

        ~ArenaManager()
        {
                // destroy device and close system
                if (m_pDevice)
                {
                        m_pSystem->DestroyDevice(m_pDevice);
                }
                Arena::CloseSystem(m_pSystem);
        }

        // update devices
        bool devicesUpdate()
        {
                m_pSystem->UpdateDevices(100);

                m_deviceList = m_pSystem->GetDevices();

                if (m_deviceList.size() == 0)
                {
                        return true;
                }
                else
                {
                        return false;
                }
        }

        // open device
        std::string OpenDevice()
        {
                if (m_pDevice)
                {
                        m_pSystem->DestroyDevice(m_pDevice);
                        m_pDevice = NULL;
                }

                m_device.clear();
                m_pSystem->UpdateDevices(100);

                m_deviceList = m_pSystem->GetDevices();
                if (m_deviceList.size() != 0)
                {
                        // grab first device from list
                        m_device = m_deviceList[0].VendorName() + " " + m_deviceList[0].ModelName() + ": Serial " + m_deviceList[0].SerialNumber();
                        m_pDevice = m_pSystem->CreateDevice(m_deviceList[0]);

                        // get nodemap
                        m_pDevice->GetNodeMap();
                }

                return m_device;
        }

        // get node map
        GenApi::INodeMap* GetNodeMap()
        {
                return m_pDevice->GetNodeMap();
        }

        // set data input to feature tree
        virtual void SetData(std::string nodeName, std::string newValue)
        {
                // get node as value
                GenApi::INodeMap* pNodeMap = m_pDevice->GetNodeMap();
                GenApi::CValuePtr pValue = pNodeMap->GetNode(nodeName.c_str());

                // set new data
                pValue->FromString(newValue.c_str());
        }
};
